`cloudfunctions/getMenu/index.js`
```js
const cloud = require('wx-server-sdk')
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })
const db = cloud.database()

exports.main = async (event) => {
  const { familyName } = event
  if (!familyName) return { success: false, msg: '缺少家庭名称' }

  try {
    const res = await db.collection('dishes')
      .where({ familyName, isAvailable: true })
      .get()

    const grouped = {}
    const order = ['主食', '汤品', '小食', '甜品']
    for (const cat of order) grouped[cat] = []
    for (const dish of res.data) {
      const cat = dish.category || '其他'
      if (!grouped[cat]) grouped[cat] = []
      grouped[cat].push(dish)
    }

    return { success: true, data: grouped }
  } catch (err) {
    console.error('getMenu failed:', err)
    return { success: false, msg: '获取菜单失败' }
  }
}
```

```js
const cloud = require('wx-server-sdk')
cloud.init({ env: cloud.DYNAMIC_CURRENT_ENV })
const db = cloud.database()

exports.main = async (event) => {
  const { familyName, items, notes } = event
  const wxContext = cloud.getWXContext()
  const openid = wxContext.OPENID

  if (!openid) return { success: false, msg: '登录已过期' }
  if (!familyName) return { success: false, msg: '缺少家庭名称' }
  if (!items || !items.length) return { success: false, msg: '购物车为空' }

  try {
    const res = await db.collection('orders').add({
      data: {
        familyName,
        dinerId: openid,
        dinerName: '',
        items,
        notes: notes || '',
        status: 'pending',
        createTime: db.serverDate()
      }
    })
    return { success: true, orderId: res._id }
  } catch (err) {
    console.error('submitOrder failed:', err)
    return { success: false, msg: '下单失败，请重试' }
  }
}
```

- [ ] **Step 4: Create menu.js**

```js
Page({
  data: {
    categories: [
      { name: '主食', icon: '🍚' },
      { name: '汤品', icon: '🍜' },
      { name: '小食', icon: '🥗' },
      { name: '甜品', icon: '🍰' }
    ],
    activeCategory: '主食',
    menuData: {},
    chefOnline: true,
    totalDishes: 0,
    cartCount: 0,
    cartTotal: 0,
    contentHeight: 600
  },

  onLoad() {
    const sys = wx.getSystemInfoSync()
    this.setData({ contentHeight: sys.windowHeight - 62 - 100 - 112 - 48 })
    this.loadMenu()
  },

  onShow() {
    this.updateCart()
  },

  loadMenu() {
    const app = getApp()
    wx.cloud.callFunction({
      name: 'getMenu',
      data: { familyName: app.globalData.familyName }
    }).then(res => {
      if (res.result.success) {
        const data = res.result.data
        let total = 0
        for (const cat of Object.keys(data)) {
          total += data[cat].length
        }
        this.setData({ menuData: data, totalDishes: total })
      }
    })
  },

  onSwitchCategory(e) {
    this.setData({ activeCategory: e.currentTarget.dataset.cat })
  },

  onAddToCart(e) {
    const dish = e.currentTarget.dataset.dish
    const app = getApp()
    const cart = app.globalData.cart || []
    const idx = cart.findIndex(item => item.dishId === dish._id)
    if (idx >= 0) {
      cart[idx].quantity += 1
    } else {
      cart.push({ dishId: dish._id, dishName: dish.name, emoji: dish.emoji, quantity: 1 })
    }
    app.globalData.cart = cart
    this.updateCart()
  },

  updateCart() {
    const cart = getApp().globalData.cart || []
    const count = cart.reduce((sum, item) => sum + item.quantity, 0)
    const total = cart.reduce((sum, item) => sum + item.quantity * 10, 0)
    this.setData({ cartCount: count, cartTotal: total })
  },

  onGoCart() {
    wx.navigateTo({ url: '/pages/family/cart/cart' })
  },

  onGoOrders() {
    wx.redirectTo({ url: '/pages/family/orders/orders' })
  }
})
```

- [ ] **Step 4: Create cart.js**

```js
Page({
  data: {
    cartItems: [],
    notes: '',
    totalPrice: 0,
    loading: false
  },

  onShow() {
    const cart = getApp().globalData.cart || []
    const total = cart.reduce((sum, item) => sum + item.quantity * 10, 0)
    this.setData({ cartItems: cart, totalPrice: total })
  },

  onChangeQty(e) {
    const { id, delta } = e.currentTarget.dataset
    const cart = getApp().globalData.cart || []
    const idx = cart.findIndex(item => item.dishId === id)
    if (idx < 0) return
    cart[idx].quantity += parseInt(delta)
    if (cart[idx].quantity <= 0) cart.splice(idx, 1)
    getApp().globalData.cart = cart
    const total = cart.reduce((sum, item) => sum + item.quantity * 10, 0)
    this.setData({ cartItems: cart, totalPrice: total })
  },

  onNotesInput(e) {
    this.setData({ notes: e.detail.value })
  },

  onClear() {
    wx.showModal({
      title: '确认清空',
      content: '确定要清空购物车吗？',
      success: (res) => {
        if (res.confirm) {
          getApp().globalData.cart = []
          this.setData({ cartItems: [], totalPrice: 0, notes: '' })
        }
      }
    })
  },

  onSubmit() {
    if (this.data.loading) return
    this.setData({ loading: true })
    const app = getApp()
    wx.cloud.callFunction({
      name: 'submitOrder',
      data: {
        familyName: app.globalData.familyName,
        items: this.data.cartItems,
        notes: this.data.notes
      }
    }).then(res => {
      if (res.result.success) {
        app.globalData.cart = []
        wx.redirectTo({ url: '/pages/family/orders/orders' })
      } else {
        wx.showToast({ title: res.result.msg || '下单失败', icon: 'none' })
        this.setData({ loading: false })
      }
    }).catch(() => {
      wx.showToast({ title: '网络错误', icon: 'none' })
      this.setData({ loading: false })
    })
  }
})
```

```js
Page({
  data: {
    orders: [],
    statusMap: { pending: '待接单', cooking: '烹饪中', ready: '已出餐' }
  },

  onShow() {
    this.loadOrders()
  },

  loadOrders() {
    const app = getApp()
    wx.cloud.callFunction({
      name: 'getOrders',
      data: { familyName: app.globalData.familyName }
    }).then(res => {
      if (res.result.success) {
        const orders = res.result.orders.map(order => ({
          ...order,
          emoji: (order.items && order.items[0] && order.items[0].emoji) || '🍽️',
          itemsSummary: order.items ? order.items.map(i => `${i.dishName} x${i.quantity}`).join('、') : '',
          timeText: order.createTime ? this.formatTime(order.createTime) : ''
        }))
        this.setData({ orders })
      }
    })
  },

  formatTime(t) {
    const d = new Date(t)
    const h = String(d.getHours()).padStart(2, '0')
    const m = String(d.getMinutes()).padStart(2, '0')
    return `${h}:${m} 下单`
  },

  onGoMenu() {
    wx.redirectTo({ url: '/pages/family/menu/menu' })
  }
})
```
